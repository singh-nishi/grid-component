export class SampleComponent {
  data: any = [];
  header: any = [];
  constructor(private _myservice: SampleService) {
     this.getRecrods();
  }
    getRecrods() {
    this._myservice.getValues().subscribe(value => {
      this.data = value;
      for (let header in this.data[0]) {
        this.header.push(header);
      }
     });
  }

}
